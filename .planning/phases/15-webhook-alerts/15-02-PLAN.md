       ```typescript
       export class DispatchAlertWebhooksService {
         constructor(private readonly deps: {
           endpointRepo: IWebhookEndpointRepository,
           deliveryRepo: IAlertDeliveryRepository,
           dispatcher: IWebhookDispatcher,
           logger?: { error: (msg: string, err?: unknown) => void },
         }) {}

         // Safe to call without await — NEVER throws (D-17 fire-and-forget safety).
         async dispatchAll(event: AlertEvent, orgName: string): Promise<void> {
           try {
             const endpoints = await this.deps.endpointRepo.findActiveByOrg(event.orgId)
             await Promise.allSettled(endpoints.map(ep => this.dispatchOne(event, orgName, ep)))
           } catch (err) {
             // Absolute safety net for fire-and-forget callers.
             this.deps.logger?.error('[DispatchAlertWebhooksService] unexpected error in dispatchAll', err)
           }
         }

         private async dispatchOne(event: AlertEvent, orgName: string, endpoint: WebhookEndpoint) {
           const dedupKey = { orgId: event.orgId, month: event.month, tier: event.tier, channel: 'webhook' as const, target: endpoint.id }
           if (await this.deps.deliveryRepo.existsSent(dedupKey)) return

           const payload = {
             orgId: event.orgId, orgName,
             tier: event.tier,
             budgetUsd: event.budgetUsd, actualCostUsd: event.actualCostUsd,
             percentage: event.percentage, month: event.month,
             alertEventId: event.id,
           }
           const dispatchedAt = new Date().toISOString()
           let result
           try {
             result = await this.deps.dispatcher.dispatch({
               url: endpoint.url,
               secret: WebhookSecret.fromExisting(endpoint.secret),
               eventType: 'alert.threshold.breached',
               payload,
             })
           } catch (err) {
             // Pitfall 4 — only network I/O is inside dispatch(); if we land here it's a programming error.
             const message = err instanceof Error ? err.message : 'Unknown dispatch error'
             const delivery = AlertDelivery.create({
               alertEventId: event.id, channel: 'webhook', target: endpoint.id,
               targetUrl: endpoint.url, dispatchedAt,
             }).markFailed(null, message, 1)
             await this.deps.deliveryRepo.save(delivery)
             return
           }
           const base = AlertDelivery.create({
             alertEventId: event.id, channel: 'webhook', target: endpoint.id,
             targetUrl: endpoint.url, dispatchedAt,
           })
           const final = result.success
             ? base.markSent(result.statusCode ?? null, new Date().toISOString(), result.attempts)
             : base.markFailed(result.statusCode ?? null, result.error ?? 'Unknown', result.attempts)
           await this.deps.deliveryRepo.save(final)
         }
       }
       ```
