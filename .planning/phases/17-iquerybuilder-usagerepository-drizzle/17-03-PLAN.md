```typescript
private translateExpression(expr: AggregateExpression): SQL {
  switch (expr.kind) {
    case 'sum':
      return sql<number>`SUM(${this.resolveColumn(expr.column)})`
    case 'count':
      return expr.column === '*'
        ? sql<number>`COUNT(*)`
        : sql<number>`COUNT(${this.resolveColumn(expr.column)})`
    case 'avg':
      return typeof expr.column === 'string'
        ? sql<number>`AVG(${this.resolveColumn(expr.column)})`
        : sql<number>`AVG(${this.translateExpression(expr.column)})`
    case 'min':
      return sql<number>`MIN(${this.resolveColumn(expr.column)})`
    case 'max':
      return sql<number>`MAX(${this.resolveColumn(expr.column)})`
    case 'dateTrunc':
      if (expr.unit !== 'day') throw new Error(`Unsupported dateTrunc unit: ${expr.unit}`)
      return sql<string>`strftime('%Y-%m-%d', ${this.resolveColumn(expr.column)})`
    case 'coalesce': {
      const [first, fallback] = expr.operands
      const firstSql = typeof first === 'string' ? this.resolveColumn(first) : this.translateExpression(first)
      return sql`COALESCE(${firstSql}, ${fallback})`
    }
    case 'add':
      return sql<number>`(${this.resolveColumn(expr.left)} + ${this.resolveColumn(expr.right)})`
    case 'column':
      return sql`${this.resolveColumn(expr.column)}`
    default: {
      const _exhaustive: never = expr
      throw new Error(`Unhandled expression kind: ${(_exhaustive as { kind: string }).kind}`)
    }
  }
}
```

4. **aggregate() method**: append to the class body:

```typescript
async aggregate<T>(spec: AggregateSpec): Promise<readonly T[]> {
  // 1. Build select object: alias → SQL expression
  const selectShape: Record<string, SQL> = {}
  for (const [alias, expr] of Object.entries(spec.select)) {
    selectShape[alias] = this.translateExpression(expr)
  }

  // 2. Build the query with accumulated where conditions
  let query = this.db.select(selectShape).from(this.table)
  if (this.conditions.length > 0) {
    query = query.where(and(...this.conditions))   // adjust to match existing where-accumulation pattern
  }

  // 3. groupBy — resolve each entry as either a select alias or a raw column
  if (spec.groupBy && spec.groupBy.length > 0) {
    const groupCols = spec.groupBy.map(name =>
      selectShape[name] !== undefined ? selectShape[name] : this.resolveColumn(name)
    )
    query = query.groupBy(...groupCols)
  }

  // 4. orderBy — same alias-or-column resolution
  if (spec.orderBy && spec.orderBy.length > 0) {
    const orderExprs = spec.orderBy.map((o: AggregateOrderBy) => {
      const base = selectShape[o.column] !== undefined ? selectShape[o.column] : this.resolveColumn(o.column)
      return o.direction === 'DESC' ? desc(base) : asc(base)
    })
    query = query.orderBy(...orderExprs)
  }

  // 5. limit
  if (spec.limit !== undefined) {
    query = query.limit(spec.limit)
  }

  const rows = await query
  return rows as readonly T[]
}
```
