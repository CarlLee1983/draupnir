Primitive semantics (MUST match Drizzle behavior):
- `sum(col)`: sum of `Number(row[col])` for all rows in group; ignores null/undefined as 0 (matches SQL SUM which skips NULL but here we must match Drizzle's observed behavior)
- `count('*')`: rows.length; `count(col)`: count of non-null rows for that column
- `avg(col)` where col is a string: (sum of non-null values) / (count of non-null values) — matches SQL AVG which skips NULL in denominator
- `avg(coalesce(col, 0))`: treat nulls as 0 first, THEN average including them in denominator — this is the Pitfall 2 case and the key difference
- `min/max(col)`: plain min/max skipping null/undefined
- `dateTrunc('day', col)`: `String(row[col]).slice(0, 10)` — stored strings are already ISO-prefixed 'YYYY-MM-DD...' so slicing gives the right bucket
- `coalesce(expr, fallback)`: evaluate expr; if result is null/undefined, return fallback
- `add(a, b)`: `Number(row[a] ?? 0) + Number(row[b] ?? 0)` — treat null as 0 (matches SQLite `x + NULL` = NULL behavior? no, SQLite returns NULL for NULL arithmetic; verify against Drizzle observed behavior. If Drizzle tests show `SUM(input_tokens + output_tokens)` returning NULL when one is NULL, Memory must match. The existing production data shouldn't have NULLs in token columns though — verify via schema: `input_tokens: integer.notNull()` means never NULL in practice.)
- `column(name)`: pass-through `row[name]`

Important:
- `add`'s NULL handling: Per schema, `input_tokens` and `output_tokens` are `notNull()`. In practice `add` never sees NULL. But for safety, Memory should match Drizzle's SQL semantics: `NULL + anything = NULL`. Implement as `(a == null || b == null) ? null : Number(a) + Number(b)`.
- After `add`, wrapping in `sum()`: SQL `SUM` skips NULL; JS `sum` with null values should also skip (or treat as 0 — verify against Drizzle observed output).

Pitfall 5 fix — operator casing:
Current `MemoryQueryBuilder.matchRow` uses `case 'LIKE'` (uppercase); `DrizzleQueryBuilder.where()` accepts lowercase `'like'`. Normalize BOTH adapters to lowercase operators and update any existing usage if tests break.

```typescript
function evalExpression(row: Record<string, unknown>, expr: AggregateExpression): unknown {
  switch (expr.kind) {
    case 'column': return row[expr.column]
    case 'dateTrunc':
      if (expr.unit !== 'day') throw new Error(`Unsupported unit: ${expr.unit}`)
      const v = row[expr.column]
      return v == null ? null : String(v).slice(0, 10)
    case 'coalesce': {
      const [first, fallback] = expr.operands
      const val = typeof first === 'string' ? row[first] : evalExpression(row, first)
      return val ?? fallback
    }
    case 'add': {
      const a = row[expr.left], b = row[expr.right]
      return (a == null || b == null) ? null : Number(a) + Number(b)
    }
    default:
      throw new Error(`evalExpression: unsupported kind '${expr.kind}' — aggregators belong in reduceAggregate`)
  }
}

function reduceAggregate(rows: readonly Record<string, unknown>[], expr: AggregateExpression): unknown {
  switch (expr.kind) {
    case 'sum': {
      let total = 0
      for (const r of rows) {
        const v = r[expr.column]
        if (v != null) total += Number(v)
      }
      return total
    }
    case 'count':
      if (expr.column === '*') return rows.length
      return rows.filter(r => r[expr.column as string] != null).length
    case 'avg': {
      // Evaluate expr.column per row (it may be a nested expression like coalesce)
      const values: number[] = []
      for (const r of rows) {
        const val = typeof expr.column === 'string' ? r[expr.column] : evalExpression(r, expr.column)
        if (val != null) values.push(Number(val))
      }
      return values.length === 0 ? 0 : values.reduce((s, x) => s + x, 0) / values.length
    }
    case 'min': {
      let m: number | null = null
      for (const r of rows) {
        const v = r[expr.column]
        if (v == null) continue
        const n = Number(v)
        if (m === null || n < m) m = n
      }
      return m
    }
    case 'max': {
      let m: number | null = null
      for (const r of rows) {
        const v = r[expr.column]
        if (v == null) continue
        const n = Number(v)
        if (m === null || n > m) m = n
      }
      return m
    }
    // Non-aggregator expressions when they appear as a select (rare): evaluate against the first row of the group (e.g. dateTrunc/column in select implicitly picks the groupBy key)
    case 'dateTrunc':
    case 'coalesce':
    case 'add':
    case 'column':
      if (rows.length === 0) return null
      return evalExpression(rows[0], expr)
  }
}
```

```typescript
async aggregate<T>(spec: AggregateSpec): Promise<readonly T[]> {
  // Use existing internal filterRows / getTableRows — DO NOT reimplement; read current private methods
  const filtered = this.applyFiltersToRows(this.getRows())
  // (exact method names depend on current MemoryQueryBuilder; adjust)

  // 1. Group
  const groups = new Map<string, Record<string, unknown>[]>()
  const groupCols = spec.groupBy ?? []
  for (const row of filtered) {
    const key = groupCols.length === 0
      ? '_all'
      : groupCols.map(g => {
          // groupBy entries can be select aliases or raw column names
          const selectExpr = spec.select[g]
          const val = selectExpr !== undefined ? evalExpression(row, selectExpr) : row[g]
          return String(val)
        }).join('\x1f')
    if (!groups.has(key)) groups.set(key, [])
    groups.get(key)!.push(row)
  }

  // 2. Evaluate select expressions per group
  let result: Record<string, unknown>[] = []
  for (const [, groupRows] of groups) {
    const out: Record<string, unknown> = {}
    for (const [alias, expr] of Object.entries(spec.select)) {
      out[alias] = reduceAggregate(groupRows, expr)
    }
    result.push(out)
  }

  // 3. orderBy
  if (spec.orderBy && spec.orderBy.length > 0) {
    result.sort((a, b) => {
      for (const o of spec.orderBy!) {
        const av = a[o.column], bv = b[o.column]
        if (av == null && bv == null) continue
        if (av == null) return o.direction === 'ASC' ? -1 : 1
        if (bv == null) return o.direction === 'ASC' ? 1 : -1
        const na = typeof av === 'number' ? av : Number(av)
        const nb = typeof bv === 'number' ? bv : Number(bv)
        if (!Number.isNaN(na) && !Number.isNaN(nb)) {
          if (na !== nb) return o.direction === 'ASC' ? na - nb : nb - na
        } else {
          const sa = String(av), sb = String(bv)
          if (sa !== sb) return o.direction === 'ASC' ? (sa < sb ? -1 : 1) : (sa < sb ? 1 : -1)
        }
      }
      return 0
    })
  }

  // 4. limit
  if (spec.limit !== undefined) {
    result = result.slice(0, spec.limit)
  }

  return result as readonly T[]
}
```
