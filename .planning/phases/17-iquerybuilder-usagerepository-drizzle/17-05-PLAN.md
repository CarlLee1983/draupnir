```typescript
/**
 * Inserts a record; silently ignores if a unique constraint on the specified target(s) already exists.
 * Maps to SQLite `INSERT ... ON CONFLICT(target) DO NOTHING` and Postgres equivalent.
 *
 * @param data - The data to insert.
 * @param options.conflictTarget - Column name (or list) that triggers the conflict-do-nothing.
 */
insertOrIgnore(data: Record<string, unknown>, options: { conflictTarget: string | readonly string[] }): Promise<void>
```

Drizzle adapter implementation shape:
```typescript
async insertOrIgnore(data, { conflictTarget }) {
  const targetCol = Array.isArray(conflictTarget)
    ? conflictTarget.map(c => (this.table as any)[c])
    : (this.table as any)[conflictTarget as string]
  await this.db.insert(this.table).values(data).onConflictDoNothing({ target: targetCol })
}
```

Memory adapter implementation shape:
```typescript
async insertOrIgnore(data, { conflictTarget }) {
  const targets = Array.isArray(conflictTarget) ? conflictTarget : [conflictTarget]
  const rows = this.getRows()
  const conflict = rows.some(r => targets.every(t => r[t] === data[t]))
  if (!conflict) rows.push(data)
}
```

```typescript
  UsageStats,

export class DrizzleUsageRepository implements IUsageRepository {
  constructor(private readonly db: IDatabaseAccess) {}

  async upsert(record: UsageRecordInsert): Promise<void> {
    await this.db.table('usage_records').insertOrIgnore(
      {
        id: record.id,
        bifrost_log_id: record.bifrostLogId,
        api_key_id: record.apiKeyId,
        org_id: record.orgId,
        model: record.model,
        provider: record.provider,
        input_tokens: record.inputTokens,
        output_tokens: record.outputTokens,
        credit_cost: record.creditCost,    // number per Plan 01

        status: record.status,
        occurred_at: record.occurredAt,
        created_at: record.createdAt,
      },
      { conflictTarget: 'bifrost_log_id' },
    )
  }

  async queryDailyCostByOrg(orgId: string, range: DateRange): Promise<readonly DailyCostBucket[]> {
    return this.db.table('usage_records')
      .where('org_id', '=', orgId)
      .whereBetween('occurred_at', [range.startDate, range.endDate])
      .aggregate<DailyCostBucket>(this.dailyCostSpec())
  }
```
