From resources/js/Pages/Member/Dashboard/Index.tsx (time window + fetch pattern):
```typescript
type WindowOption = 7 | 30 | 90
const WINDOW_OPTIONS = [{ value: 7, label: '7d' }, { value: 30, label: '30d' }, { value: 90, label: '90d' }]
const DAY_MS = 24 * 60 * 60 * 1000

function resolveDateRange(window: WindowOption) {
  const endTime = new Date().toISOString()
  const startTime = new Date(Date.now() - (window - 1) * DAY_MS).toISOString()
  return { startTime, endTime }
}

// useEffect with AbortController for cancellable fetch
useEffect(() => {
  if (!orgId) { setLoading(false); return }
  const controller = new AbortController()
  const { startTime, endTime } = resolveDateRange(selectedWindow)
  const query = new URLSearchParams({ start_time: startTime, end_time: endTime })
  fetch(`/api/organizations/${orgId}/dashboard/per-key-cost?${query}`, { signal: controller.signal })
    .then(r => r.json())
    .then(payload => { if (payload.success) setData(payload.data) })
    .catch(err => { if (err.name !== 'AbortError') setFetchError(String(err)) })
    .finally(() => setLoading(false))
  return () => controller.abort()
}, [orgId, selectedWindow])
```
