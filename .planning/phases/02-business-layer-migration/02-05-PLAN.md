```typescript
async getStats(
  virtualKeyIds: readonly string[],
  query?: UsageQuery,
): Promise<UsageStats> {
  if (virtualKeyIds.length === 0) {
    return { totalRequests: 0, totalCost: 0, totalTokens: 0, avgLatency: 0 }
  }

  const stats = await this.gatewayClient.getUsageStats(virtualKeyIds, query)
  // Stats are already camelCase from interface — return directly
  return {
    totalRequests: stats.totalRequests,
    totalCost: stats.totalCost,
    totalTokens: stats.totalTokens,
    avgLatency: stats.avgLatency,
  }
}
```

Change `getLogs()` signature and body:
```typescript
async getLogs(
  virtualKeyIds: readonly string[],
  query?: UsageQuery,
): Promise<readonly LogEntry[]> {
  if (virtualKeyIds.length === 0) {
    return []
  }

  return this.gatewayClient.getUsageLogs(virtualKeyIds, query)
}
```

Note: The old `response.logs` property access (from `bifrostClient.getLogs()`) is gone — `getUsageLogs` returns `LogEntry[]` directly.
