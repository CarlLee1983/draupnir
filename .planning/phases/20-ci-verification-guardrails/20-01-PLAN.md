        }
      }

      const failures = results.filter(r => !r.ok)
      if (failures.length > 0) {
        console.error(`\n❌ DI audit failed — ${failures.length}/${results.length} tokens unresolvable:`)
        const grouped = { missing: [] as typeof failures, circular: [] as typeof failures, 'factory-failed': [] as typeof failures }
        for (const f of failures) grouped[f.category!].push(f)
        for (const [cat, items] of Object.entries(grouped)) {
          if (items.length === 0) continue
          console.error(`\n  [${cat}] ${items.length}:`)
          for (const item of items) console.error(`    - ${item.token}: ${item.error}`)
        }
        process.exit(1)
      }
      console.log(`✅ DI audit passed — ${results.length} bindings resolvable`)
      process.exit(0)
    }
