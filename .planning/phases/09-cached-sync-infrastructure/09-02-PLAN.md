```typescript
/**
 * IUsageRepository — Dashboard Read Model Port
 *
 * Application-layer port for local usage_records queries.
 * Implemented by DrizzleUsageRepository in Infrastructure.
 * Used by Phase 10 chart services instead of live Bifrost API calls.
 *
 * Constraint: All fields readonly (immutability rule).
 * Constraint: credit_cost stored as TEXT in DB — callers receive number (repository converts).
 */

export interface DateRange {
  readonly startDate: string  // ISO 8601 datetime string
  readonly endDate: string    // ISO 8601 datetime string
}

/** One day's aggregated cost and usage — used by cost trend AreaChart */
export interface DailyCostBucket {
  readonly date: string              // 'YYYY-MM-DD'
  readonly totalCost: number         // SUM(CAST(credit_cost AS REAL))
  readonly totalRequests: number     // COUNT(*)
  readonly totalInputTokens: number  // SUM(input_tokens)
  readonly totalOutputTokens: number // SUM(output_tokens)
}

/** Per-model aggregation — used by model comparison BarChart and table */
export interface ModelUsageBucket {
  readonly model: string
  readonly provider: string | null
  readonly totalCost: number       // SUM(CAST(credit_cost AS REAL))
  readonly totalRequests: number   // COUNT(*)
  readonly avgLatencyMs: number    // AVG(latency_ms) — may be 0 if column null
}

/** Org-level or key-level aggregate stats — used by KPI summary cards */
export interface UsageStats {
  readonly totalRequests: number
  readonly totalCost: number
  readonly totalTokens: number    // SUM(input_tokens + output_tokens)
  readonly avgLatency: number     // AVG(latency_ms)
}

/** Shape used by BifrostSyncService when upserting a new record */
export interface UsageRecordInsert {
  readonly id: string              // crypto.randomUUID()
  readonly bifrostLogId: string    // LogEntry.logId ?? `${log.timestamp}:${log.keyId}`
  readonly apiKeyId: string        // resolved local api_keys.id
  readonly orgId: string           // resolved from ApiKey.orgId
  readonly model: string
  readonly provider: string | null
  readonly inputTokens: number
  readonly outputTokens: number
  readonly creditCost: string      // TEXT decimal string e.g. '0.00123'
  readonly latencyMs: number | null
  readonly status: 'success' | 'error' | null
  readonly occurredAt: string      // ISO 8601
  readonly createdAt: string       // ISO 8601
}

export interface IUsageRepository {
  /** Upsert a usage record. Idempotent — UNIQUE constraint on bifrost_log_id. */
  upsert(record: UsageRecordInsert): Promise<void>

  /** Daily cost aggregation for an org. Used by cost trend chart (DASHBOARD-02). */
  queryDailyCostByOrg(orgId: string, range: DateRange): Promise<readonly DailyCostBucket[]>

  /** Per-model breakdown. Used by model comparison chart + table (DASHBOARD-03, DASHBOARD-05). */
  queryModelBreakdown(orgId: string, range: DateRange): Promise<readonly ModelUsageBucket[]>

  /** Aggregate KPI stats for an entire org. Used by summary cards (DASHBOARD-01). */
  queryStatsByOrg(orgId: string, range: DateRange): Promise<UsageStats>

  /** Aggregate KPI stats scoped to one API key. Used for MEMBER role scoping. */
  queryStatsByKey(apiKeyId: string, range: DateRange): Promise<UsageStats>
}
```

**Create `src/Modules/Dashboard/Application/Ports/ISyncCursorRepository.ts`:**

```typescript
/**
 * ISyncCursorRepository — Sync Cursor Persistence Port
 *
 * Tracks the high-water mark for incremental Bifrost log sync.
 * The cursor_type 'bifrost_logs' is the primary cursor used by BifrostSyncService.
 *
 * Implemented by DrizzleSyncCursorRepository in Infrastructure.
 */

export interface SyncCursor {
  readonly cursorType: string
  readonly lastSyncedAt: string | null   // ISO 8601; null = never synced
  readonly lastBifrostLogId: string | null
}

export interface ISyncCursorRepository {
  /** Retrieve the current cursor for the given cursorType. Returns null if never set. */
  get(cursorType: string): Promise<SyncCursor | null>

  /** Advance the cursor after a successful sync batch. Upserts if not yet set. */
  advance(cursorType: string, update: { readonly lastSyncedAt: string; readonly lastBifrostLogId?: string }): Promise<void>
}
```

**Create `src/Modules/Dashboard/Application/DTOs/UsageLogDTO.ts`:**

This DTO is the validated shape that BifrostSyncService receives from `ILLMGatewayClient.getUsageLogs()` before inserting into `usage_records`. It documents the expected shape of `LogEntry` as consumed by the sync service.

```typescript
/**
 * UsageLogDTO — Validated shape of a LogEntry as consumed by BifrostSyncService.
 *
 * LogEntry comes from ILLMGatewayClient.getUsageLogs() — already camelCase,
 * already gateway-neutral. BifrostSyncService maps LogEntry → UsageRecordInsert
 * (after resolving apiKeyId via ApiKeyRepository.findByBifrostVirtualKeyId).
 *
 * This DTO documents the mapping, not a transformation layer.
 * No Zod validation here — LogEntry is typed by the gateway interface.
 */

export interface UsageLogDTO {
  /** Stable unique ID from the Bifrost log entry. May be undefined for legacy logs. */
  readonly logId: string | undefined
  /** Bifrost virtual key ID (NOT the local api_keys.id — must be resolved). */
  readonly keyId: string
  readonly model: string
  readonly provider: string
  readonly inputTokens: number
  readonly outputTokens: number
  readonly latencyMs: number
  readonly cost: number
  readonly status: 'success' | 'error'
  readonly timestamp: string  // ISO 8601
}

/**
 * Maps a LogEntry from ILLMGatewayClient.getUsageLogs() to UsageLogDTO.
 * Provides a stable conversion function for BifrostSyncService to use.
 */
export function toUsageLogDTO(entry: {
  readonly logId?: string
  readonly keyId: string
  readonly model: string
  readonly provider: string
  readonly inputTokens: number
  readonly outputTokens: number
  readonly latencyMs: number
  readonly cost: number
  readonly status: 'success' | 'error'
  readonly timestamp: string
}): UsageLogDTO {
  return {
    logId: entry.logId,
    keyId: entry.keyId,
    model: entry.model,
    provider: entry.provider,
    inputTokens: entry.inputTokens,
    outputTokens: entry.outputTokens,
    latencyMs: entry.latencyMs,
    cost: entry.cost,
    status: entry.status,
    timestamp: entry.timestamp,
  }
}
```
